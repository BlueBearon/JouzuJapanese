    /**
     * Returns the ConjugationType from the string
     * @param text
     * @return
     */
    public static ConjugationType fromString(String text) {
        if (text != null) {
            for (ConjugationType b : ConjugationType.values()) {
                if (text.equalsIgnoreCase(b.toString())) {
                    return b;
                }
            }
        }
        return null;
    }
